#include <stdio.h>
#include <stdlib.h>

/*
write a program to check whether a given number is positive or negative
*/

int main()
{
    int a = 212;
    if (a > 0) {
        printf("%d is positive", a);
    }
    else if (a == 0) {
        printf("%d is equal to 0", a);
    }
    else {
        printf("%d is negative", a);
    }
    return 0;
}
